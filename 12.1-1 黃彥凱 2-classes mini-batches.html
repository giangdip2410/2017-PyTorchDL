class Net(torch.nn.Module):
    def __init__(self, n_feature, n_hidden, n_output,n_hidden1):
        super(Net, self).__init__()
        self.a = torch.nn.Linear(n_feature, n_hidden)
        self.b = torch.nn.Linear(n_hidden, n_hidden1)
        self.c = torch.nn.Linear(n_hidden1, n_output)
    
    def forward(self, x):
        x = F.relu(self.a(x))
        x = F.relu(self.b(x))
        x = self.c(x)
        return x
net = Net(2, 5, 6, 8)

for epoch in range(100):   # train entire dataset 3 times
    for step, (batch_x, batch_y) in enumerate(loader):  # for each training step
        b_x=Variable(batch_x)
        b_y=Variable(batch_y)
        out = net(b_x)
        loss = loss_func(out, b_y)
        losslst.append(loss.data.numpy()[0])
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        prediction = torch.max(F.softmax(out, 1), 1)[1]   
        pred_y = prediction.data.numpy().squeeze()
        target_y = batch_y.numpy()
        accuracy = sum(pred_y == target_y)/400
        acculst.append(accuracy)
